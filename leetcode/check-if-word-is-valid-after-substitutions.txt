/*
 * Problem: 1003. Check If Word Is Valid After Substitutions
 * URL: https://leetcode.com/problems/check-if-word-is-valid-after-substitutions/
 * Platform: leetcode
 */

class Solution:
    def isValid(self, s: str) -> bool:
        s1 = ""
        while s1 != s:
            s, s1 = s.replace("abc", ""), s
        return s == ""
